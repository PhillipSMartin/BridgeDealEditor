<script>
  function setCardScore(element, leftNumber, rightNumber) {
    element.style.setProperty("--left-number", `"${leftNumber}"`);
    element.style.setProperty("--right-number", `"${rightNumber}"`);
    element.setAttribute("data-left", leftNumber);
    element.setAttribute("data-right", rightNumber);
    element.setAttribute("aria-label", `Score: ${leftNumber} to ${rightNumber}`);
  }

  const score = document.querySelector(".card-score");

  // Example:
  setCardScore(score, 1, 0);

  // Other examples:
  // setCardScore(score, 7, 3);
  // setCardScore(score, "10", "12");
</script>
